Ex 1

i. Que livros são do autor com o nome Mia Couto?

SELECT l.nome AS titulo
FROM livro l
JOIN autor a 
    ON l.autor_id = a.autor_id
WHERE a.nome = 'Mia'
  AND a.apelido = 'Couto';

ii. Quantos livros existem?

SELECT COUNT(*) AS total_livros
FROM livro;


iii. Quantos livros existem por autor?

SELECT a.nome,
       a.apelido,
       COUNT(l.livro_id) AS total_livros
FROM autor a
LEFT JOIN livro l 
       ON a.autor_id = l.autor_id
GROUP BY a.autor_id, a.nome, a.apelido;


iv. Quais os livros favoritos do utilizador BrunaSilva?

SELECT l.livro_id,
       l.nome AS titulo,
       l.genero,
       l.ano_publicacao
FROM favorito f
JOIN utilizador u 
       ON f.utilizador_id = u.utilizador_id
JOIN livro l 
       ON f.livro_id = l.livro_id
WHERE u.username = 'Bruna Silva';


v. Que utilizadores não têm livros favoritos?

SELECT u.utilizador_id,
       u.username
FROM utilizador u
LEFT JOIN favorito f 
       ON u.utilizador_id = f.utilizador_id
WHERE f.livro_id IS NULL;


vi. Que géneros (sem duplicados) de livros existem?

SELECT DISTINCT genero
FROM livro;


vii. Que utilizadores têm mais do que dois livros favoritos, ordenado de forma
inversa ao alfabeto?

SELECT u.utilizador_id,
       u.username,
       COUNT(f.livro_id) AS total_favoritos
FROM utilizador u
JOIN favorito f 
       ON u.utilizador_id = f.utilizador_id
GROUP BY u.utilizador_id, u.username
HAVING COUNT(f.livro_id) > 2
ORDER BY u.username DESC;
